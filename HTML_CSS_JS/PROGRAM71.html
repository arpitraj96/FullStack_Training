// Define an asynchronous function that simulates fetching data from an API async function fetchDataFromAPI(): Promise { return new Promise((resolve, reject) => { // Simulate an API request with a delay setTimeout(() => { const data = "Data from API"; resolve(data); }, 2000); // Simulate a 2-second delay }); } // Define an asynchronous function that uses the fetched data async function processData() { try { console.log("Fetching data..."); const data = await fetchDataFromAPI(); console.log(`Data received: ${data}`); } catch (error) { console.error(`Error: ${error}`); } } // Call the asynchronous function to process data processData(); console.log("Processing data asynchronously...");
